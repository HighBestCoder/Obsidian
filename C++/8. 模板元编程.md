
# trivial与standard-layout

```C++
#include <iostream>

namespace test_trait {

template<typename T, T v>
struct integral_constant {
    static constexpr T value = v;
};

using true_type = integral_constant<bool, true>;
using false_type = integral_constant<bool, false>;

template<typename T>
struct is_trivial : integral_constant<bool, __is_trivial(T)> {
};

template<typename T>
struct is_standard_layout : integral_constant<bool, __is_standard_layout(T)> {
};

template<typename T>
struct is_pod : integral_constant<bool, __is_trivial(T) && __is_standard_layout(T)> {
};

/// 这个是c++14提供的功能，就是可以在使用的时候，省略test_trait::is_pod<T>::value
// 而是直接像下面这样使用
template<typename T>
inline constexpr bool is_pod_v = is_pod<T>::value;

} // end namespace test_trait

int main(void) {
    std::cout << test_trait::is_pod<int>::value << std::endl;
    std::cout << test_trait::is_pod_v<double> << std::endl;
    return 1;
}
```
